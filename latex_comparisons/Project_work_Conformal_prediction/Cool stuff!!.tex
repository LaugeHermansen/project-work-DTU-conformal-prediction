\begin{gather}
    EI(x)=\begin{cases}
    (\mu(\mathbf{x})-f(\mathbf{x}^+))\Phi(Z)+\sigma(\mathbf{x})\phi(Z) & \text{if }\sigma(\mathbf{x})>0 \\
    0 & \text{if }\sigma(\mathbf{x})\leq 0
    \end{cases}
\end{gather}
